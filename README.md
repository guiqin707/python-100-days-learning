## 💻 今日代码示例

from greenlet import greenlet
def sing():
    print('在唱歌')
    g2.switch()
    print('唱完歌了')
def dance():
    print('在跳舞')
    print('跳完舞了')
if __name__ == '__main__':
    #创建协程对象 greenlet(任务名)
    g1 = greenlet(sing)
    g2 = greenlet(dance)
    g1.switch()#切换到g1中运行
    g2.switch()
