# 匹配网址    前缀一般是www，后缀：.com、.cn、.org
li = ['www.baidu.com','www.python.org','http.jd.cn','www.py.en','www.abc.cn']
# res = re.match(r'www\.\w*\.(com|cn|org)','www.baidu.com')
# print(res.group())
for i in li:
    res = re.match(r'www(\.)\w*\1(com|cn|org)',i)
    if res != None:#如果 res 不等于 None
        print(res.group())
    else:
        print(f'{i}这个网址有错误')
# \1 是反向引用，表示"匹配与第一个分组完全相同的内容"
# 在你的例子中，第一个分组是 (\.)，所以 \1 也必须匹配一个点号
# 因此 \1 和 (\.) 实际上是等价的
