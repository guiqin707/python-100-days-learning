# 实例属性
# 格式：
# self.属性名
# 实例属性是实例对象特有的，类属性可以大家都访问，但是实例属性不可以，为私有
class Person:
    name = 'guigui'
    def introduce(self):
        print(f'我的名字是{self.name},我的年龄是{self.age}')
pe = Person()
pe.age = 18
pe.introduce()
# 构造函数
# __init__()
# 通常用来做属性初始化或者赋值操作
# 注意：在类实例化对象的时候会自动被调用
class Person:
    def __init__(self, name, age,hobby):
        self.name = name
        self.age = age
        self.hobby = hobby
    def introduce(self):
        print(f'我的名字是{self.name},年龄是{self.age},我的爱好是{self.hobby}')
    def play(self):
        print(f'{self.name}喜欢{self.hobby}')
pe = Person('guigui',18,'玩地铁逃生当囤囤党')#传参，每一次实例化对象都要传参
pe.introduce()
pe.play()
